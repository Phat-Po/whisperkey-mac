    ```python
    from ApplicationServices import (
        AXUIElementCreateSystemWide,
        AXUIElementCopyAttributeValue,
        kAXFocusedUIElementAttribute,
        kAXRoleAttribute,
        kAXTextFieldRole,
        kAXTextAreaRole,
        kAXComboBoxRole,
        kAXErrorSuccess,
        # NOTE: kAXSearchFieldRole does NOT exist — verified by ImportError in venv
    )

    _TEXT_INPUT_ROLES: frozenset[str] = frozenset({
        kAXTextFieldRole,
        kAXTextAreaRole,
        kAXComboBoxRole,
        "AXSearchField",  # kAXSearchFieldSubrole == 'AXSearchField'; no kAXSearchFieldRole
    })

    def is_cursor_in_text_field() -> bool:
        try:
            system_wide = AXUIElementCreateSystemWide()
            err, focused = AXUIElementCopyAttributeValue(
                system_wide, kAXFocusedUIElementAttribute, None
            )
            if err != kAXErrorSuccess or focused is None:
                return False
            err2, role = AXUIElementCopyAttributeValue(focused, kAXRoleAttribute, None)
            if err2 != kAXErrorSuccess or role is None:
                return False
            return role in _TEXT_INPUT_ROLES
        except Exception:
            return False
    ```
